function Factorial(number) {
for (var loop = number - 1; loop > 1; loop--) {
number *= loop
}

return number;
}

function FizBuz(fizA, fizB) {
var buzA = "";
for (var i = 1; i <= 100; i++) {
if (i % fizA == 0 && i % fizB == 0) {
buzA += "FizzBuzz, ";
} else if (i % fizA == 0) {
buzA += "Fizz, ";
} else if (i % fizB == 0) {
buzA += "Buzz, ";
} else buzA += i + ", ";

}
return buzA;
}

$("#palCheck").click(function () {
var input = $("#palInput").val();
if (ispal(input)) {
$("#palOutput").text(input + " is a Palindrome!");

}
else {
$("#palOutput").text(input + " is NOT a Palindrome!");
}
});

function ispal(palinput) {
var output = "";
for (var i = palinput.length - 1; i >= 0; i--) {
output += palinput[i];
}
return palinput === output;
}
